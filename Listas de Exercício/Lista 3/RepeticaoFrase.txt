/* 2. Faça um programa que escreve na tela a mesma frase 10 vezes. E depois faça com que o programa mostre o número de cada linha no início e no final da linha */

public class RepeticaoFrase {
    public static void main(String[] args) {

        for (int i=0; i < 11; i++) {
            System.out.println(i + " Sou um programa Java! " + i);
        }
    }
}
